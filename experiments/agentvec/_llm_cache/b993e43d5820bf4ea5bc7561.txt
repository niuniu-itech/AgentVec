#include <riscv_vector.h>
#include <math.h>

void layernorm(int n, const float *x, float *y) {
    float s = 0;
    for (int i = 0; i < n; i++) s += x[i];
    float mean = s / n;
    float v = 0;
    for (int i = 0; i < n; i++) {
        float d = x[i] - mean;
        v += d * d;
    }
    float var = v / n;
    float rstd = 1.0f / sqrtf(var + 1e-5f);
    int vl = __riscv_vsetvl_e32m8(n);
    int i = 0;
    for (; i + vl <= n; i += vl) {
        vfloat32m8_t vx = vle32_v_f32_m(vl, x + i);
        vfloat32m8_t vy = vmul_vf_f32_m(vl, vx, mean, vl);
        vy = vsub_vf_f32_m(vl, vx, mean, vl);
        vy = vmul_vf_f32_m(vl, vy, rstd, vl);
        vse32_v_f32_m(vl, y + i, vy);
    }
    for (; i < n; i++) {
        y[i] = (x[i] - mean) * rstd;
    }
}
