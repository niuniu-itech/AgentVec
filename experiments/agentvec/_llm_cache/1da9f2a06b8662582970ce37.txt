#include <riscv_vector.h>
#include <math.h>

void agentvec_kernel(const float *a, const float *b, const float *c, float *out, int n) {
    if (n <= 0) {
        return;
    }

    float sum = 0.0f;
    int vl;
    vfloat32m1_t va, vsum;
    vfloat32m1_t vm1 = vmv_v_f32_m1(0.0f);

    // Strip-mining loop for vector-length-agnostic execution
    for (int i = 0; i < n; i += vl) {
        vl = __riscv_vsetvl_e32m1(n - i);
        
        // Load vector from a
        va = vle32_v_f32_m1(&a[i], vl);
        
        // Multiply element-wise by itself (a * a)
        vfloat32m1_t va_sq = vmul_vv_f32_m1(va, va, vl);
        
        // Accumulate into sum vector
        vsum = vadd_vv_f32_m1(vsum, va_sq, vl);
    }

    // Reduce the vector sum to a scalar
    // Since vsum is m1, we can use vreduce-add
    // However, vreduce functions return the element type.
    // We need to handle the case where the loop didn't run or vl=0 properly.
    // Initialize vsum to 0. If n > 0, the loop runs at least once with vl >= 1.
    
    // Re-calculate sum using a standard reduction approach compatible with m1
    // vle32_v_f32_m1 returns a masked vector. We need to sum all active elements.
    // A simple way for m1 is to use vreduce-add if available, or manual reduction.
    // RVV 1.0 has vreduce-add.
    
    // Let's restart the accumulation logic to be cleaner.
    // We need a scalar accumulator.
    
    float total_sum = 0.0f;
    
    // Reset for the actual computation
    // We can accumulate into a vector register and then reduce at the end of each chunk,
    // or accumulate into a scalar. Given m1 is small, reducing per chunk is efficient.
    
    for (int i = 0; i < n; i += vl) {
        vl = __riscv_vsetvl_e32m1(n - i);
        
        vfloat32m1_t va = vle32_v_f32_m1(&a[i], vl);
        vfloat32m1_t va_sq = vmul_vv_f32_m1(va, va, vl);
        
        // Reduce the vector chunk to a scalar
        float chunk_sum = __riscv_vreduce_add_f32_m1(va_sq);
        total_sum += chunk_sum;
    }

    out[0] = sqrtf(total_sum);
}
