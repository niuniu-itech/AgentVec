void gemm(int N, const double *A, const double *B, double *C) {
    // C[i][j] = sum_k A[i][k] * B[k][j]
    // We process row i of C, computing all N elements.
    // For each row i, we need to compute C[i][j] for j=0..N-1
    // C[i][j] = dot product of A[i,:] and B[:,j]
    
    // We'll use strip-mining over the reduction dimension (k)
    // and vectorize over the output dimension (j)
    
    for (int i = 0; i < N; i++) {
        // For each output element C[i][j], initialize to 0
        // We'll accumulate in vector registers
        
        // Process in chunks of VL elements over j
        int j = 0;
        while (j < N) {
            vltype_t vl = vsetvl_e64m1(N - j);
            vfloat64m1_t vsum = vfmv_v_f_f64m1(0.0, vl);
            
            // Strip-mine over k dimension
            for (int k = 0; k < N; k++) {
                // Load A[i][k] as a scalar, broadcast to vector
                double a_ik = A[i * N + k];
                vfloat64m1_t va = vfmv_v_f_f64m1(a_ik, vl);
                
                // Load B[k][j..j+vl-1]
                vfloat64m1_t vb = vle64_v_f64m1(B + k * N + j, vl);
                
                // Multiply and accumulate
                vsum = vfmacc_vv_f64m1(vsum, va, vb, vl);
            }
            
            // Store result
            vse64_v_f64m1(C + i * N + j, vsum, vl);
            
            j += vl;
        }
    }
}
