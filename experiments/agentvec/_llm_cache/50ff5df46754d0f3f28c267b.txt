#include <riscv_vector.h>

void gemv(int N, const double *A, const double *x, double *y) {
    for (int i = 0; i < N; ++i) {
        const double *a_row = A + i * N;
        double sum = 0.0;
        int j = 0;
        vfloat64m1_t vsum = __riscv_vle64_v_f64m1(&sum, 1);
        vsum = __riscv_vmv_v_v_f64m1(vsum, 0, 1);

        while (j < N) {
            size_t vl = __riscv_vsetvl_e64m1(N - j);
            vfloat64m1_t va = __riscv_vle64_v_f64m1(a_row + j, vl);
            vfloat64m1_t vx = __riscv_vle64_v_f64m1(x + j, vl);
            vsum = __riscv_vfmacc_vv_f64m1(vsum, vx, va, vl);
            j += vl;
        }

        vfloat64m1_t vred = __riscv_vfmv_v_f64m1(0.0, 1);
        vred = __riscv_vfredusum_vs_f64m1_f64m1(vsum, vred, N);
        y[i] = __riscv_vfmv_f_s_f64m1_f64(vred);
    }
}
