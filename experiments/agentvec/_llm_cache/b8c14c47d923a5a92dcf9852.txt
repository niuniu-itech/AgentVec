#include <riscv_vector.h>

void kern(int n, double a, const double *x, double *y) {
    int i = 0;
    // Set vector length to maximum available (up to n), using element type double (e64) and group m8
    vuint64m8_t vvl = __riscv_vsetvl_e64m8(n);
    // Broadcast scalar 'a' to a vector register
    vfloat64m8_t va = __riscv_vfmv_v_f_f64m8(a, vvl);

    // Main vector loop (strip-mining)
    while (i < n) {
        // Determine the number of elements to process in this iteration
        vuint64m8_t vl = __riscv_vsetvl_e64m8(n - i);

        // Load vectors from memory
        vfloat64m8_t vx = __riscv_vle64_v_f64m8(x + i, vl);
        vfloat64m8_t vy = __riscv_vle64_v_f64m8(y + i, vl);

        // Fused multiply-add: vy = va * vx + vy
        vy = __riscv_vfmadd_vv_f64m8(vy, va, vx, vl);

        // Store result
        __riscv_vse64_v_f64m8(y + i, vy, vl);

        // Increment pointer
        i += __riscv_vmv_x_s_u64m8_u64(vl);
    }
}
