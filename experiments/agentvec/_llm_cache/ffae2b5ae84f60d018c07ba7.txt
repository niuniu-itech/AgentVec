#include <riscv_vector.h>
#include <math.h>

void layernorm(int n, const float *x, float *y) {
    float mean = 0.0f, var = 0.0f;
    int vl;
    
    // Compute sum for mean
    vfloat32m8_t vsum = __riscv_vfmv_v_f_f32m8(0.0f, __riscv_vsetvl_e32m8(n));
    for (int i = 0; i < n; i += vl) {
        vl = __riscv_vsetvl_e32m8(n - i);
        vfloat32m8_t vx = __riscv_vle32_v_f32m8(x + i, vl);
        vsum = __riscv_vfadd_vv_f32m8(vsum, vx, vl);
    }
    float sum = __riscv_vfmv_f_s_f32m8_f32(__riscv_vfredusum_vs_f32m8_f32m1(vsum, __riscv_vfmv_v_f_f32m1(0.0f, 1), __riscv_vsetvl_e32m8(1)));
    mean = sum / n;
    
    // Compute variance
    vfloat32m8_t vvar = __riscv_vfmv_v_f_f32m8(0.0f, __riscv_vsetvl_e32m8(n));
    for (int i = 0; i < n; i += vl) {
        vl = __riscv_vsetvl_e32m8(n - i);
        vfloat32m8_t vx = __riscv_vle32_v_f32m8(x + i, vl);
        vfloat32m8_t vdiff = __riscv_vfsub_vf_f32m8(vx, mean, vl);
        vvar = __riscv_vfmacc_vv_f32m8(vvar, vdiff, vdiff, vl);
    }
    float variance = __riscv_vfmv_f_s_f32m8_f32(__riscv_vfredusum_vs_f32m8_f32m1(vvar, __riscv_vfmv_v_f_f32m1(0.0f, 1), __riscv_vsetvl_e32m8(1)));
    var = variance / n;
    
    // Compute rstd and apply normalization
    float rstd = 1.0f / sqrtf(var + 1e-5f);
    for (int i = 0; i < n; i += vl) {
        vl = __riscv_vsetvl_e32m8(n - i);
        vfloat32m8_t vx = __riscv_vle32_v_f32m8(x + i, vl);
        vfloat32m8_t vdiff = __riscv_vfsub_vf_f32m8(vx, mean, vl);
        vfloat32m8_t vy = __riscv_vfmul_vf_f32m8(vdiff, rstd, vl);
        __riscv_vse32_v_f32m8(y + i, vy, vl);
    }
}
