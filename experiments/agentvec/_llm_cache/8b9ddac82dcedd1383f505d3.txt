#include <math.h>
#include <riscv_vector.h>

void agentvec_kernel(const float *a, const float *b, const float *c, float *out, int n) {
    (void)b;
    (void)c;
    float s = 0;
    size_t vl;
    const float *ptr = a;

    // Use a vector accumulator to sum absolute values
    vfloat32m1_t vsum = __riscv_vfmv_v_f_f32m1(0.0f, 1);

    while (n > 0) {
        vl = __riscv_vsetvl_e32m1(n);
        vfloat32m1_t va = __riscv_vle32_v_f32m1(ptr, vl);
        vfloat32m1_t vabs_a = __riscv_vfabs_v_f32m1(va, vl);
        vsum = __riscv_vfadd_vv_f32m1(vsum, vabs_a, vl);
        ptr += vl;
        n -= vl;
    }

    // Reduce the vector sum to a scalar
    s = __riscv_vfmv_f_s_f32m1_f32(vsum);

    out[0] = s;
}
