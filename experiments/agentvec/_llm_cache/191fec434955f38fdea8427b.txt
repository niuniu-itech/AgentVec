#include <riscv_vector.h>
#include <math.h>

void agentvec_kernel(const float *a, const float *b, const float *c, float *out, int n) {
    // Compute sum of squares of a
    vfloat32m1_t vec_zero = __riscv_vfmv_v_f_f32m1(0.0f, __riscv_vsetvl_e32m1(n));
    vfloat32m1_t sum_vec = vec_zero;
    int vl;
    int i = 0;
    while (i < n) {
        vl = __riscv_vsetvl_e32m1(n - i);
        vfloat32m1_t vec_a = __riscv_vle32_v_f32m1(&a[i], vl);
        vfloat32m1_t sq = __riscv_vfmul_vv_f32m1(vec_a, vec_a, vl);
        sum_vec = __riscv_vfadd_vv_f32m1(sum_vec, sq, vl);
        i += vl;
    }
    // Reduce sum to scalar
    vfloat32m1_t sum_reduced = __riscv_vfredusum_vs_f32m1_f32m1(sum_vec, vec_zero, vl);
    float sum_sq;
    __riscv_vse32_v_f32m1(&sum_sq, sum_reduced, vl);
    out[0] = sqrtf(sum_sq);
}
