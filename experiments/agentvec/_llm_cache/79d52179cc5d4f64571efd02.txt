#include <riscv_vector.h>

void gemm(int N, const double *A, const double *B, double *C) {
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            const double *a_ptr = &A[i * N];
            const double *b_ptr = &B[j];
            size_t vl;
            vfloat64m1_t v_sum = __riscv_vfmv_v_f_f64m1(0.0, 1);

            // Vector-length-agnostic strip-mining loop for the dot product
            for (int k = 0; k < N; k += vl) {
                vl = __riscv_vsetvl_e64m1(N - k);

                // Load vector from A (contiguous)
                vfloat64m1_t v_a = __riscv_vle64_v_f64m1(a_ptr, vl);

                // Load vector from B (strided by N)
                vfloat64m1_t v_b = __riscv_vlse64_v_f64m1(b_ptr, sizeof(double) * N, vl);

                // Accumulate: v_sum += v_a * v_b
                v_sum = __riscv_vfmacc_vv_f64m1(v_sum, v_a, v_b, vl);

                a_ptr += vl;
                b_ptr += vl;
            }

            // Reduce the vector sum to a scalar
            double sum = __riscv_vfmv_f_s_f64m1_f64(v_sum);
            vfloat64m1_t v_tmp = __riscv_vslideup_vx_f64m1(v_sum, v_sum, 1, 1);
            sum += __riscv_vfmv_f_s_f64m1_f64(v_tmp);

            // Handle reduction for remaining elements if LMUL > 1 (though m1 is requested)
            // Since we use m1, the loop above covers all elements.
            // However, to be robust for the requested reduction logic:
            // A standard reduction for m1 would look like this:
            vfloat64m1_t v_red = __riscv_vfmv_v_f_f64m1(0.0, 1);
            v_red = __riscv_vfredusum_vs_f64m1_f64m1(v_sum, v_red, __riscv_vsetvl_e64m1(N));
            sum = __riscv_vfmv_f_s_f64m1_f64(v_red);

            C[i * N + j] = sum;
        }
    }
}
