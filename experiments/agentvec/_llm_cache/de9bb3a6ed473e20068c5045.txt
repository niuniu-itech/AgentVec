#include <riscv_vector.h>
#include <math.h>

void agentvec_kernel(const float *a, const float *b, const float *c, float *out, int n) {
    (void)b;
    (void)c;
    
    float s = 0.0f;
    int i = 0;
    vfloat32m1_t va, vsq;
    vfloat32m1_t vsum = vmv_v_f_f32m1_m(vsetzero_v_f32m1(), 0.0f, 0);
    
    for (; i + 31 < n; i += 32) {
        size_t vl = __riscv_vsetvl_e32m1(n - i);
        va = vle32_v_f32m1(a + i, vl);
        vsq = vmul_vv_f32m1_m(vsum, va, va, vl);
        vsum = vsq;
    }
    
    // Reduce the vector sum to a scalar
    // We need to sum all elements in vsum. Since we used m1, we can reduce it.
    // However, standard reduction might not be directly available as a single intrinsic for all VLENs in a simple way without loops or specific intrinsics.
    // A safe way for m1 is to use vreduce_add_vv or similar if available, but typically we might need to reduce manually or use vreduce_add.
    // In RVV 1.0, vreduce_add_v_f32m1 exists.
    
    s = vreduce_add_v_f32m1(vsum);
    
    // Handle remaining elements
    for (; i < n; i++) {
        s += a[i] * a[i];
    }
    
    out[0] = sqrtf(s);
}
