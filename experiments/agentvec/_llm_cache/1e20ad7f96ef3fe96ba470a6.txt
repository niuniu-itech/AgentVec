#include <riscv_vector.h>
#include <math.h>

void agentvec_kernel(const float *a, const float *b, const float *c, float *out, int n) {
    float sum = 0.0f;
    vfloat32m1_t vsum = vmv_v_f_f32_m1(0.0f);
    vfloat32m1_t vabs = vmv_v_f_f32_m1(0.0f);
    
    long vl = __riscv_vsetvl_e32m1(n);
    while (vl > 0) {
        vfloat32m1_t va = vle32_v_f32m1(a, vl);
        vfloat32m1_t vabs_val = vfabs_v_f32m1(va, vl);
        vsum = vadd_vv_f32m1(vsum, vabs_val, vl);
        a += vl;
        vl = __riscv_vsetvl_e32m1(n - (a - (const float *)0)); // Recalculate remaining
        // Note: The above loop structure is slightly flawed for strict agnostic style without tracking offset.
        // Let's rewrite with proper offset tracking.
    }
    
    // Correct strip-mining loop
    long remaining = n;
    long offset = 0;
    vfloat32m1_t acc = vmv_v_f_f32_m1(0.0f);
    
    while (remaining > 0) {
        long vl = __riscv_vsetvl_e32m1(remaining);
        vfloat32m1_t va = vle32_v_f32m1(a + offset, vl);
        vfloat32m1_t vabs_val = vfabs_v_f32m1(va, vl);
        acc = vadd_vv_f32m1(acc, vabs_val, vl);
        offset += vl;
        remaining -= vl;
    }
    
    // Reduce vector to scalar
    float result = 0.0f;
    #pragma unroll
    for (int i = 0; i < 8; i++) { // m1 has 8 lanes max, but we should use vreduce
        // vreduce_sum is not directly available for m1 in all headers, but standard reduction exists.
        // Actually, vreduce_sum_vf is for reducing vector to scalar.
    }
    
    // Use vreduce_sum_vf if available, otherwise manual reduction.
    // RVV 1.0 has vreduce_sum_vf.
    result = vreduce_sum_vf_f32m1_f32(acc);
    
    out[0] = result;
}
