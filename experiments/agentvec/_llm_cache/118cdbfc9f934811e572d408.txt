#include <riscv_vector.h>
#include <math.h>

void agentvec_kernel(const float *a, const float *b, const float *c, float *out, int n) {
    (void)b;
    (void)c;
    float s = 0.0f;
    int vl;
    for (int i = 0; i < n; i += vl) {
        vl = __riscv_vsetvl_e32m1(n - i);
        vfloat32m1_t vec_a = __riscv_vle32_v_f32m1(a + i, vl);
        vfloat32m1_t prod = __riscv_vfmul_vv_f32m1(vec_a, vec_a, vl);
        vfloat32m1_t red = __riscv_vfredusum_vs_f32m1_f32m1(prod, __riscv_vfmv_v_f_f32m1(0.0f, vl), vl);
        s += __riscv_vfmv_f_s_f32m1_f32(red);
    }
    out[0] = sqrtf(s);
}
