#include <riscv_vector.h>

void gemm(int N, const double *A, const double *B, double *C) {
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            double sum = 0.0;
            const double *a_row = A + i * N;
            const double *b_col = B + j;
            int k = 0;
            while (k < N) {
                int vl = __riscv_vsetvl_e64m1(N - k);
                vfloat64m1_t va = __riscv_vle64_v_f64m1(a_row + k, vl);
                vfloat64m1_t vb = __riscv_vle64_v_f64m1(b_col + k * N, vl);
                vfloat64m1_t vprod = __riscv_vfmul_vv_f64m1(va, vb, vl);
                vfloat64m1_t vsum = __riscv_vfmv_v_f_f64m1(0.0, vl);
                vsum = __riscv_vfmacc_vv_f64m1(vsum, va, vb, vl);
                sum += __riscv_vfmv_f_s_f64m1_f64(__riscv_vfredusum_vs_f64m1_f64m1(vsum, __riscv_vfmv_v_f_f64m1(0.0, vl), vl));
                k += vl;
            }
            C[i * N + j] = sum;
        }
    }
}
